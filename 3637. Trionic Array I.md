class Solution(object):
    def isTrionic(self, nums):
        """
        :type nums: List[int]
        :rtype: bool
        """
        foundp = False
        foundq = False
        if nums[0]==nums[1]:
                    return False
        if nums[0]> nums[1]:
            return False
        for i in range(1,len(nums)-1):
            if nums[i]==nums[i+1]:
                    return False
            if not foundp:
                if nums[i]>nums[i+1] and not foundp:
                    foundp=True
            elif foundp and not foundq:
                if nums[i]<nums[i+1] and not foundq:
                    foundq=True
            elif foundp and foundq:

                    return False
        if foundp and foundq:
            return True
