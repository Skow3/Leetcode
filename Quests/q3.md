# Max consecutive ones:

# Solution 1:
python
```python
class Solution(object):
    def findMaxConsecutiveOnes(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        ans=0
        t=0
        for i in nums:
            if i==1:
                t+=1
            else:
                ans=max(t,ans)
                t=0
        ans=max(t,ans)
        return ans
```

```python
class Solution {
public:
    int findMaxConsecutiveOnes(vector<int>& nums) {
        int ans =0;
        int t=0;
        for(int i=0;i<nums.size();i++){
            if(nums[i]==1){
                t+=1;
            }else{
                if(ans<t){
                    ans=t;
                }
                t=0;
            }
        }
        if(ans<t){
            ans=t;
        }
        return ans;
    }
};
```
