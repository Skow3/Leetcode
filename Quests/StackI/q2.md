# Evaluate Reverse Polish Notation
MEDIUM

* The question was easy as we already did something like this in TOC.
* Main thing was to truncating towards zero after each operation.

# SOLUTION 1 
PYTHON3
```python
class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        p=[]
        for i in range(len(tokens)):
            if tokens[i]=="+":
                n = p.pop()
                m = p.pop()
                p.append(m+n)
            elif tokens[i]=="/":
                n = p.pop()
                m = p.pop()
                p.append(int(m/n))
            elif tokens[i]=="*":
                n = p.pop()
                m = p.pop()
                p.append(m*n)
            elif tokens[i]=="-":
                n = p.pop()
                m = p.pop()
                p.append(m-n)
            else:
                p.append(int(tokens[i]))
        return p.pop()
```

class Solution(object):
    def evalRPN(self, tokens):
        p = []
        for t in tokens:
            if t == "+":
                b = p.pop()
                a = p.pop()
                p.append(a + b)
            elif t == "-":
                b = p.pop()
                a = p.pop()
                p.append(a - b)
            elif t == "*":
                b = p.pop()
                a = p.pop()
                p.append(a * b)
            elif t == "/":
                b = p.pop()
                a = p.pop()
                p.append(int(float(a) / b))  # key fix
            else:
                p.append(int(t))
        return p.pop()
```

class Solution {
public:
    int evalRPN(vector<string>& tokens) {
        vector<int> p;

        for (int i = 0; i < tokens.size(); i++) {
            if (tokens[i] == "+") {
                int n = p.back(); p.pop_back();
                int m = p.back(); p.pop_back();
                p.push_back(m + n);
            }
            else if (tokens[i] == "-") {
                int n = p.back(); p.pop_back();
                int m = p.back(); p.pop_back();
                p.push_back(m - n);
            }
            else if (tokens[i] == "*") {
                int n = p.back(); p.pop_back();
                int m = p.back(); p.pop_back();
                p.push_back(m * n);
            }
            else if (tokens[i] == "/") {
                int n = p.back(); p.pop_back();
                int m = p.back(); p.pop_back();
                p.push_back(m / n); 
            }
            else {
                p.push_back(stoi(tokens[i]));
            }
        }
        return p.back();
    }
};
```
