# Build an Array With Stack Operations
Medium

```python
class Solution(object):
    def buildArray(self, target, n):
        """
        :type target: List[int]
        :type n: int
        :rtype: List[str]
        """
        c=0
        ans=[]
        for i in range(1,n+1):
            if c>=len(target):
                break
            ans.append("Push")
            if target[c]==i:
                c+=1
            else:
                ans.append("Pop")
        return ans

```


# C++
```c++
class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
        vector<string> ans;
        int c=0;
        for(int i=1;i<n+1;i++){
            if(c>=target.size()){
                break;
            }
            ans.push_back("Push");
            if(target[c]==i){
                c++;
            }else{
                ans.push_back("Pop");
            }
        }
        return ans;
    }
};
```
