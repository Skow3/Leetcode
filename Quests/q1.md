```python
class Solution(object):

        """
        :type nums: List[int]
        :rtype: List[int]
        """
        ans=[]
        for i in nums:
            ans.append(i)
        for i in nums:
            ans.append(i)
        return ans
```

```python
class Solution(object):
    def getConcatenation(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        ans=[0]*(2*len(nums))
        for i in range(len(nums)):
            ans[i]=nums[i]
            ans[i+len(nums)]=nums[i]
        return ans
```

# C++
```cpp
class Solution {
public:
    vector<int> getConcatenation(vector<int>& nums) {
        int n=nums.size();
        vector<int> ans(2*n);
        for(int i=0;i<n;i++){
            ans[i]=nums[i];
            ans[i+n]=nums[i];
        }
        return ans;
    }
};
```
