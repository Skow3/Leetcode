```python
class Solution(object):
    def counter(self,nums,c):
        ct=0
        for i in nums:
            if i<c:
                ct+=1
            else:
                break
        return ct
    def smallerNumbersThanCurrent(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        s= sorted(nums)
        ans=[]
        for i in nums:
            ans.append(self.counter(s,i))
        return ans
```

# Solution 2:
```cpp
class Solution {
public:
    int counter(vector<int>& nums,int c){
        int ct=0;
        for(int i=0;i<nums.size();i++){
            if(nums[i]<c){
                ct+=1;
            }
            else{
                break;
            }
        }
        return ct;
    }
    vector<int> smallerNumbersThanCurrent(vector<int>& nums) {
        vector<int> s= nums;
        sort(s.begin(),s.end());
        vector<int> ans;
        ans.reserve(nums.size());
        for(int i=0;i<nums.size();i++){
            ans.push_back(counter(s,nums[i]));
        }
        return ans;
    }
};
```
