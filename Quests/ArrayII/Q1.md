class Solution(object):
    def findErrorNums(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        s= sum(nums)
        m= len(nums)
        dub=0
        ans=[]
        for i in sorted(nums):
            if dub==i:
                break
            else:
                dub=i
        nsum= (m*(m+1))/2
        a= dub - (s-nsum)
        ans.append(dub)
        ans.append(a)
        return ans

```cpp
class Solution {
public:
    vector<int> findErrorNums(vector<int>& nums) {
        vector<int> ans;
        ans.reserve(2);
        int s=0; // SUM
        for(int i=0;i<nums.size();i++){
            s+=nums[i];
        }
        int m = nums.size();
        int nsum = 0 ;// TOTAL ELEMENT SUM
        int dub=0;
        sort(nums.begin(), nums.end());
        for(int i=0;i<nums.size();i++){
            if(dub==nums[i]){
                break;
            }
            else{
                dub=nums[i];
            }
        }
        nsum= (m*(m+1))/2;
        int a= dub-(s-nsum);
        ans.push_back(dub);
        ans.push_back(a);
        return ans;
    }
};
```

## What i learnt :
* So i was trying to sort the nums vector using sort but i forgot that to sort in cpp we need to give iterators not the vector itself so

  Instead of
  ```cpp
  sort(nums);
  ```
  It should be:
  ```cpp
  sort(nums.begin(),nums.end());
  ```
