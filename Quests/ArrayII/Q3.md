# Solution Optimized:
```python
class Solution(object):
    def findDisappearedNumbers(self, nums):
        # Mark visited numbers
        for i in range(len(nums)):
            idx = abs(nums[i]) - 1
            if nums[idx] > 0:
                nums[idx] = -nums[idx]

        # Collect missing numbers
        ans = []
        for i in range(len(nums)):
            if nums[i] > 0:
                ans.append(i + 1)

        return ans
```
